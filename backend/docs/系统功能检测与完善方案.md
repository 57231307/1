#### 1.1.2 核心算法设计

**折旧计算方法**:
```rust
// 平均年限法
fn straight_line_depreciation(
    original_value: Decimal,
    salvage_value: Decimal,
    useful_life_months: i32,
) -> Decimal {
    (original_value - salvage_value) / Decimal::from(useful_life_months)
}

// 双倍余额递减法
fn double_declining_balance(
    book_value: Decimal,
    useful_life_months: i32,
    accumulated_depreciation: Decimal,
) -> Decimal {
    let rate = Decimal::from(2) / Decimal::from(useful_life_months);
    book_value * rate
}

// 年数总和法
fn sum_of_years_digits(
    depreciable_base: Decimal,
    remaining_life: i32,
    total_life: i32,
) -> Decimal {
    let sum_of_years = total_life * (total_life + 1) / 2;
    depreciable_base * Decimal::from(remaining_life) / Decimal::from(sum_of_years)
}
```

    fn current_ratio(current_assets: Decimal, current_liabilities: Decimal) -> Decimal {
        current_assets / current_liabilities
    }
    
    /// 速动比率
    fn quick_ratio(
        current_assets: Decimal,
        inventory: Decimal,
        current_liabilities: Decimal,
    ) -> Decimal {
        (current_assets - inventory) / current_liabilities
    }
    
    /// 资产负债率
    fn debt_to_asset_ratio(total_liabilities: Decimal, total_assets: Decimal) -> Decimal {
        total_liabilities / total_assets
    }
    
    /// 净资产收益率 (ROE)
    fn roe(net_profit: Decimal, average_equity: Decimal) -> Decimal {
        net_profit / average_equity

impl SupplierEvaluationService {
    /// 综合评分计算
    pub fn calculate_total_score(
        quality_score: Decimal,      // 质量得分（权重 40%）
        delivery_score: Decimal,     // 交货得分（权重 30%）
        price_score: Decimal,        // 价格得分（权重 20%）
        service_score: Decimal,      // 服务得分（权重 10%）
    ) -> Decimal {
        quality_score * Decimal::from(0.4)
            + delivery_score * Decimal::from(0.3)
            + price_score * Decimal::from(0.2)
            + service_score * Decimal::from(0.1)
    }
    
    /// 等级评定
    pub fn evaluate_grade(total_score: Decimal) -> SupplierGrade {
        if total_score >= Decimal::from(90) {
            SupplierGrade::A  // 优秀供应商
        } else if total_score >= Decimal::from(80) {
            SupplierGrade::B  // 良好供应商
        } else if total_score >= Decimal::from(70) {
            SupplierGrade::C  // 合格供应商
        } else if total_score >= Decimal::from(60) {
            SupplierGrade::D  // 需改进供应商
        } else {
            SupplierGrade::E  // 不合格供应商
        }
