### 4. 库存调整 - 单号生成

```rust
async fn generate_adjustment_no(&self) -> Result<String, DbErr> {
    let now = Utc::now();
    let date_str = now.format("%Y%m%d").to_string();
    
    // 查询当天最大单号
    let max_no = inventory_adjustment::Entity::find()
        .filter(
            inventory_adjustment::Column::AdjustmentNo
                .like(format!("ADJ{}%", date_str))
        )
        .order_by_desc(inventory_adjustment::Column::AdjustmentNo)
        .one(&self.db)
        .await?;

    let seq = match max_no {
        Some(model) => {
            let no_str = model.adjustment_no.replace(&format!("ADJ{}", date_str), "");
            no_str.parse::<u32>().unwrap_or(0) + 1
        }
        None => 1,
    };

    Ok(format!("ADJ{}{:04}", date_str, seq))
}
```

## 数据库表结构

### 库存调整单表

```sql
CREATE TABLE inventory_adjustments (
    id SERIAL PRIMARY KEY,
    adjustment_no VARCHAR(50) NOT NULL UNIQUE,      -- 调整单号
    warehouse_id INTEGER NOT NULL,                   -- 仓库 ID
    adjustment_date TIMESTAMPTZ NOT NULL,            -- 调整日期
    adjustment_type VARCHAR(20) NOT NULL,            -- 调整类型
    reason_type VARCHAR(20) NOT NULL,                -- 原因类型
    reason_description TEXT,                         -- 原因说明
    total_quantity DECIMAL(12,2) NOT NULL,           -- 总数量
    notes TEXT,                                      -- 备注
    created_by INTEGER,                              -- 创建人
    approved_by INTEGER,                             -- 审核人
    approved_at TIMESTAMPTZ,                         -- 审核时间
    status VARCHAR(20) NOT NULL,                     -- 状态
    created_at TIMESTAMPTZ NOT NULL,
    updated_at TIMESTAMPTZ NOT NULL
);
```

### 库存调整明细表

```sql
CREATE TABLE inventory_adjustment_items (
    id SERIAL PRIMARY KEY,
    adjustment_id INTEGER NOT NULL,                  -- 调整单 ID
    stock_id INTEGER NOT NULL,                       -- 库存 ID
    quantity DECIMAL(12,2) NOT NULL,                 -- 调整数量
    quantity_before DECIMAL(12,2) NOT NULL,          -- 调整前数量
    quantity_after DECIMAL(12,2) NOT NULL,           -- 调整后数量
    unit_cost DECIMAL(12,2),                         -- 单位成本
    amount DECIMAL(12,2),                            -- 调整金额
    notes TEXT,                                      -- 备注
    created_at TIMESTAMPTZ NOT NULL,
    updated_at TIMESTAMPTZ NOT NULL
);
```

### 操作日志表

```sql
CREATE TABLE operation_logs (
    id SERIAL PRIMARY KEY,
    user_id INTEGER,                                  -- 用户 ID
    username VARCHAR(50),                             -- 用户名
    module VARCHAR(50) NOT NULL,                      -- 模块
    action VARCHAR(20) NOT NULL,                      -- 操作类型
    description TEXT,                                 -- 操作描述
    request_method VARCHAR(10),                       -- 请求方法
    request_uri VARCHAR(200),                         -- 请求 URI
    request_ip VARCHAR(50),                           -- 请求 IP
    user_agent TEXT,                                  -- 用户代理
    status VARCHAR(20) NOT NULL,                      -- 状态
    error_message TEXT,                               -- 错误信息
    duration_ms BIGINT,                               -- 耗时（毫秒）
    extra_data JSONB,                                 -- 额外数据
    created_at TIMESTAMPTZ NOT NULL
);
```
