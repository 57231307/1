/// 创建供应商（含事务）
pub async fn create(
    &self,
    req: CreateSupplierRequest,
    user_id: i32,
) -> Result<Supplier, AppError> {
    let txn = self.db.begin().await?;
    
    // 1. 生成供应商编码
    let supplier_code = self.generate_supplier_code().await?;
    
    // 2. 创建供应商
    let supplier = suppliers::ActiveModel {
        supplier_code: Set(supplier_code),
        supplier_name: Set(req.supplier_name),
        supplier_short_name: Set(req.supplier_short_name),
        // ... 其他字段
        created_by: Set(Some(user_id)),
        ..Default::default()
    }.insert(&txn).await?;
    
    // 3. 创建联系人
    for contact_req in req.contacts {
        supplier_contacts::ActiveModel {
            supplier_id: Set(supplier.id),
            contact_name: Set(contact_req.contact_name),
            // ... 其他字段
        }.insert(&txn).await?;
    }
    
    // 4. 创建资质
    for qual_req in req.qualifications {
        supplier_qualifications::ActiveModel {
            supplier_id: Set(supplier.id),
            qualification_name: Set(qual_req.qualification_name),
            // ... 其他字段
        }.insert(&txn).await?;
    }
    
    txn.commit().await?;
    Ok(supplier)
}
```

/// 计算评估分数
pub fn calculate_total_score(scores: &EvaluationScores) -> Decimal {
    // 质量水平（35 分）
    let quality_score = {
        let pass_rate_score = if scores.quality_pass_rate >= 98.0 {
            dec!(15.0)
        } else if scores.quality_pass_rate >= 95.0 {
            dec!(12.0)
        } else if scores.quality_pass_rate >= 90.0 {
            dec!(8.0)
        } else {
            Decimal::ZERO
        };
        
        let incident_score = match scores.quality_incidents {
            0 => dec!(10.0),
            1 => dec!(5.0),
            _ => Decimal::ZERO,
        };
        
        pass_rate_score + incident_score + scores.quality_improvement_score
    };
    
    // 交货能力（25 分）
    let delivery_score = {
        let on_time_score = if scores.delivery_on_time_rate >= 95.0 {
            dec!(15.0)
        } else if scores.delivery_on_time_rate >= 90.0 {
            dec!(12.0)
        } else {
            Decimal::ZERO
        };
        
        let completion_score = if scores.order_completion_rate >= 98.0 {
            dec!(10.0)
        } else if scores.order_completion_rate >= 95.0 {
            dec!(8.0)
        } else {
            Decimal::ZERO
        };
        
        on_time_score + completion_score
    };
    
    // ... 价格、服务、技术得分
    
    quality_score + delivery_score + price_score + service_score + tech_score
}
```
