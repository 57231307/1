#### 1.3.1 批次成本核算

```rust
pub struct BatchCostCalculator {
    // 成本归集
    pub direct_material: Decimal,    // 直接材料
    pub direct_labor: Decimal,       // 直接人工
    pub manufacturing_overhead: Decimal, // 制造费用
    pub processing_fee: Decimal,     // 委托加工费
}

impl BatchCostCalculator {
    // 计算批次总成本
    pub fn calculate_total_cost(&self) -> Decimal {
        self.direct_material + self.direct_labor + 
        self.manufacturing_overhead + self.processing_fee
    }
    
    // 计算单位成本
    pub fn calculate_unit_cost(
        &self,
        output_quantity: Decimal
    ) -> Option<Decimal> {
        if output_quantity.is_zero() {
            None
        } else {
            Some(self.calculate_total_cost() / output_quantity)
        }
    }
}
```

#### 1.3.2 色号利润分析

```rust
pub struct ColorNoProfitAnalyzer {
    // 按色号归集销售和成本
    pub color_sales: HashMap<String, ColorSalesData>,
}

pub struct ColorSalesData {
    pub color_no: String,
    pub sales_quantity: Decimal,  // 销售米数
    pub sales_revenue: Decimal,   // 销售收入
    pub sales_cost: Decimal,      // 销售成本
}

impl ColorNoProfitAnalyzer {
    // 计算毛利润
    pub fn calculate_gross_profit(&self, color_no: &str) -> Option<Decimal> {
        self.color_sales.get(color_no)
            .map(|data| data.sales_revenue - data.sales_cost)
    }
    
    // 计算毛利率
    pub fn calculate_gross_margin(&self, color_no: &str) -> Option<Decimal> {
        self.color_sales.get(color_no)
            .and_then(|data| {
                if data.sales_revenue.is_zero() {
                    None
                } else {
                    Some((data.sales_revenue - data.sales_cost) / data.sales_revenue)
                }
            })
    }
}
```

#### 1.3.3 双计量单位换算

```rust
pub struct DualUnitConverter {
    pub primary_unit: String,      // 主单位：米
    pub secondary_unit: String,    // 辅单位：公斤
    pub conversion_rate: Decimal,  // 换算率：克重（克/米）
}

impl DualUnitConverter {
    // 米数转公斤数
    pub fn meters_to_kg(&self, meters: Decimal) -> Decimal {
        // 公斤数 = 米数 × 克重 / 1000
        meters * self.conversion_rate / Decimal::from(1000)
    }
    
    // 公斤数转米数
    pub fn kg_to_meters(&self, kg: Decimal) -> Decimal {
        // 米数 = 公斤数 × 1000 / 克重
        kg * Decimal::from(1000) / self.conversion_rate
    }
}
```

```rust
    // 计算批次成本
    pub async fn calculate_batch_cost(
        &self,
        batch_no: &str,
        period: &str,
    ) -> Result<BatchCostResult, AppError> {
        // 1. 归集直接材料
        let direct_material = self
            .collect_direct_material(batch_no, period)
            .await?;
        
        // 2. 归集直接人工
        let direct_labor = self
            .collect_direct_labor(batch_no, period)
            .await?;
        
        // 3. 归集制造费用
        let overhead = self
            .collect_manufacturing_overhead(batch_no, period)
            .await?;
        
        // 4. 归集委托加工费
        let processing_fee = self
            .collect_processing_fee(batch_no, period)
            .await?;
        
        // 5. 计算总成本和单位成本
        let total_cost = direct_material + direct_labor + overhead + processing_fee;
        let output_quantity = self.get_output_quantity(batch_no, period).await?;
        let unit_cost = if output_quantity.is_zero() {
            Decimal::ZERO
        } else {
            total_cost / output_quantity
        };
        
        Ok(BatchCostResult {
            batch_no: batch_no.to_string(),
            direct_material,
            direct_labor,
            overhead,
            processing_fee,
            total_cost,
            output_quantity,
            unit_cost,
        })
    }

        &self,
        batch_no: &str,

pub struct ColorNoProfitService {
    db: Arc<DatabaseConnection>,

impl ColorNoProfitService {
    pub fn new(db: Arc<DatabaseConnection>) -> Self {
        Self { db }
    }
    
    // 分析色号利润
    pub async fn analyze_color_profit(
        &self,
        period: &str,
    ) -> Result<Vec<ColorNoProfitAnalysis>, AppError> {
        // 1. 查询各色号的销售收入
        let sales = self.get_color_sales(period).await?;
        
        // 2. 查询各色号的销售成本
        let costs = self.get_color_costs(period).await?;
        
        // 3. 计算毛利润和毛利率
        let mut analyses = Vec::new();
        for (color_no, sales_data) in sales {
            let cost_data = costs.get(&color_no).copied().unwrap_or(Decimal::ZERO);
            let gross_profit = sales_data - cost_data;
            let gross_margin = if sales_data.is_zero() {
                Decimal::ZERO
            } else {
                gross_profit / sales_data
            };
            
            analyses.push(ColorNoProfitAnalysis {
                color_no,
                sales_revenue: sales_data,
                sales_cost: cost_data,
                gross_profit,
                gross_margin,
            });
        }
        
        // 4. 按毛利率排序
        analyses.sort_by(|a, b| b.gross_margin.cmp(&a.gross_margin));
        
        Ok(analyses)
    }
    
    async fn get_color_sales(
        &self,
        period: &str,
    ) -> Result<HashMap<String, Decimal>, AppError>;
    
    async fn get_color_costs(
        &self,
        period: &str,
    ) -> Result<HashMap<String, Decimal>, AppError>;
}
```

## 九、技术难点与解决方案

### 9.1 技术难点

#### 难点 1：批次成本核算的准确性

**问题**: 面料行业批次成本构成复杂（坯布成本 + 染料成本 + 染费 + 人工 + 制造费用）

**解决方案**:
- 采用事务确保成本归集完整性
- 分步骤计算，每步可追溯
- 提供成本计算日志

#### 难点 2：双计量单位换算精度

**问题**: 米数和公斤数换算涉及小数精度

**解决方案**:
- 使用 Rust Decimal 类型确保精度
- 统一换算率（克重）存储
- 所有计算保留 4 位小数，显示 2 位

#### 难点 3：凭证过账的并发控制

**问题**: 多用户同时过账可能导致余额计算错误

**解决方案**:
- 使用数据库事务
- 悲观锁控制科目余额表
- 凭证号按月份连续编号

### 9.2 性能优化

- 科目树缓存（Redis）
- 余额表按月分表
- 异步处理报表生成
