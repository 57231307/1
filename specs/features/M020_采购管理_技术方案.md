## 四、核心业务逻辑

### 4.1 采购申请转订单流程
```rust
pub async fn convert_requisition_to_order(
    &self,
    requisition_id: i32,
    supplier_id: i32,
) -> Result<purchase_order::Model, DbErr> {
    let txn = self.db.begin().await?;
    
    // 1. 查询采购申请单
    let requisition = PurchaseRequisition::find_by_id(requisition_id)
        .one(&txn)
        .await?
        .ok_or(DbErr::RecordNotFound("采购申请单不存在"))?;
    
    // 2. 检查申请单状态
    if requisition.status != "已审核" {
        return Err(DbErr::Custom("只有已审核的申请单才能转订单"));
    }
    
    // 3. 查询申请单明细
    let items = PurchaseRequisitionItem::find()
        .filter(purchase_requisition_item::Column::RequisitionId.eq(requisition_id))
        .all(&txn)
        .await?;
    
    // 4. 创建采购订单
    let order_no = self.generate_order_no();
    let order = purchase_order::ActiveModel {
        order_no: Set(order_no),
        supplier_id: Set(supplier_id),
        purchase_type: Set(requisition.material_type),
        status: Set("待审核"),
        // ... 其他字段
        ..Default::default()
    };
    
    let created_order = order.insert(&txn).await?;
    
    // 5. 创建订单明细（复制申请单明细，增加采购价格）
    for (idx, item) in items.into_iter().enumerate() {
        let order_item = purchase_order_item::ActiveModel {
            order_id: Set(created_order.id),
            line_no: Set((idx + 1) as i32),
            product_id: Set(item.product_id),
            quantity_ordered: Set(item.quantity_requested),
            // 复制面料行业字段
            yarn_count: Set(item.yarn_count),
            density: Set(item.density),
            width_cm: Set(item.width_cm),
            gram_weight: Set(item.gram_weight),
            // ... 其他字段
            ..Default::default()
        };
        order_item.insert(&txn).await?;
    }
    
    // 6. 更新申请单状态
    let mut req: purchase_requisition::ActiveModel = requisition.into();
    req.status = Set("已转订单".to_string());
    req.update(&txn).await?;
    
    // 7. 提交事务
    txn.commit().await?;
    
    Ok(created_order)
}
```

### 4.2 采购入库流程（带质检）
```rust
pub async fn create_receipt_with_inspection(
    &self,
    receipt_input: CreateReceiptInput,
) -> Result<purchase_receipt::Model, DbErr> {
    let txn = self.db.begin().await?;
    
    // 1. 查询采购订单
    let order = PurchaseOrder::find_by_id(receipt_input.purchase_order_id)
        .one(&txn)
        .await?
        .ok_or(DbErr::RecordNotFound("采购订单不存在"))?;
    
    // 2. 创建入库单
    let receipt_no = self.generate_receipt_no();
    let receipt = purchase_receipt::ActiveModel {
        receipt_no: Set(receipt_no),
        purchase_order_id: Set(receipt_input.purchase_order_id),
        warehouse_id: Set(receipt_input.warehouse_id),
        status: Set("待质检"),
        // ... 其他字段
        ..Default::default()
    };
    
    let created_receipt = receipt.insert(&txn).await?;
    
    // 3. 创建入库明细
    let mut total_quantity = Decimal::ZERO;
    let mut qualified_quantity = Decimal::ZERO;
    let mut rejected_quantity = Decimal::ZERO;
    
    for (idx, item) in receipt_input.items.into_iter().enumerate() {
        // 质检判断
        let inspection_result = if item.actual_gram_weight.is_some() 
            && item.actual_width_cm.is_some() {
            "合格"
        } else {
            "不合格"
        };
        
        if inspection_result == "合格" {
            qualified_quantity += item.quantity_received;
        } else {
            rejected_quantity += item.quantity_received;
        }
        total_quantity += item.quantity_received;
        
        // 创建库存记录（面料行业特色：批次 + 色号 + 缸号 + 等级）
        self.create_stock_with_fabric_features(
            &txn,
            item.product_id,
            receipt_input.warehouse_id,
            item.batch_no.clone(),
            item.color_no.clone(),
            item.dye_lot_no.clone(),
            item.grade.clone(),
            item.quantity_received,
            item.actual_gram_weight,
            item.actual_width_cm,
        ).await?;
        
        // 创建入库明细
        let receipt_item = purchase_receipt_item::ActiveModel {
            receipt_id: Set(created_receipt.id),
            line_no: Set((idx + 1) as i32),
            quantity_received: Set(item.quantity_received),
            batch_no: Set(item.batch_no),
            color_no: Set(item.color_no),
            dye_lot_no: Set(item.dye_lot_no),
            grade: Set(item.grade),
            inspection_result: Set(inspection_result.to_string()),
            actual_width_cm: Set(item.actual_width_cm),
            actual_gram_weight: Set(item.actual_gram_weight),
            // ... 其他字段
            ..Default::default()
        };
        receipt_item.insert(&txn).await?;
    }
    
    // 4. 更新入库单汇总
    let mut receipt_model: purchase_receipt::ActiveModel = created_receipt.into();
    receipt_model.total_quantity = Set(total_quantity);
    receipt_model.qualified_quantity = Set(qualified_quantity);
    receipt_model.rejected_quantity = Set(rejected_quantity);
    if rejected_quantity == Decimal::ZERO {
        receipt_model.status = Set("质检合格".to_string());
    }
    receipt_model.update(&txn).await?;
    
    // 5. 更新采购订单的已到货数量
    self.update_order_received_quantity(&txn, receipt_input.purchase_order_id, total_quantity).await?;
    
    // 6. 提交事务
    txn.commit().await?;
    
    Ok(receipt_model.into_model().unwrap())
}
```

/// 双计量单位换算（米 ↔ 公斤）
pub fn convert_quantity(
    quantity_meters: Decimal,
    gram_weight: Option<Decimal>,
) -> Result<Decimal, String> {
    match gram_weight {
        Some(gw) => {
            // 米数 × 克重 (g/m²) × 幅宽 (m) ÷ 1000 = 公斤数
            // 简化：假设幅宽为标准幅宽
            let width_m = Decimal::from(1.8); // 默认 180cm
            let quantity_kg = quantity_meters * gw * width_m / Decimal::from(1000);
            Ok(quantity_kg)
        }
        None => Err("缺少克重信息，无法换算".to_string())
    }
}

/// 正向追溯：供应商 → 采购 → 库存 → 生产 → 销售 → 客户
pub async fn forward_trace(
    &self,
    batch_no: &str,
    color_no: &str,
) -> Result<TraceResult, DbErr> {
    // 1. 查询采购入库
    let receipts = PurchaseReceiptItem::find()
        .filter(purchase_receipt_item::Column::BatchNo.eq(batch_no))
        .filter(purchase_receipt_item::Column::ColorNo.eq(color_no))
        .all(&self.db)
        .await?;
    
    // 2. 查询库存
    let stocks = InventoryStock::find()
        .filter(inventory_stock::Column::BatchNo.eq(batch_no))
        .filter(inventory_stock::Column::ColorNo.eq(color_no))
        .all(&self.db)
        .await?;
    
    // 3. 查询生产领用
    let production_usage = ProductionMaterialUsage::find()
        .filter(production_material_usage::Column::BatchNo.eq(batch_no))
        .filter(production_material_usage::Column::ColorNo.eq(color_no))
        .all(&self.db)
        .await?;
    
    // 4. 查询销售出库
    let sales_delivery = SalesDeliveryItem::find()
        .filter(sales_delivery_item::Column::BatchNo.eq(batch_no))
        .filter(sales_delivery_item::Column::ColorNo.eq(color_no))
        .all(&self.db)
        .await?;
    
    Ok(TraceResult {
        receipts,
        stocks,
        production_usage,
        sales_delivery,
    })
}
