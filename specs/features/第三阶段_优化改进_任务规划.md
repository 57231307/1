/// 内存健康检查（详细版）
fn check_memory() -> HealthCheckItem {
    let sys = System::new_all();
    let total_mem = sys.total_memory();
    let used_mem = sys.used_memory();
    let available_mem = sys.available_memory();
    let usage_rate = used_mem as f64 / total_mem as f64;
    
    let message = format!(
        "内存：{:.1}GB / {:.1}GB (使用率：{:.1}%, 可用：{:.1}GB)",
        used_mem as f64 / 1024.0 / 1024.0,
        total_mem as f64 / 1024.0 / 1024.0,
        usage_rate * 100.0,
        available_mem as f64 / 1024.0 / 1024.0
    );
    
    let status = if usage_rate > 0.9 {
        "unhealthy"
    } else if usage_rate > 0.7 {
        "degraded"
    } else {
        "healthy"
    }.to_string();
    
    HealthCheckItem {
        status,
        message: Some(message),
        response_time_ms: None,
    }
}

/// 磁盘健康检查（详细版）
fn check_disk() -> HealthCheckItem {
    let disks = Disks::new_with_refreshed_list();
    let mut total_disk = 0u64;
    let mut used_disk = 0u64;
    
    for disk in disks.list() {
        total_disk += disk.total_space();
        used_disk += disk.total_space() - disk.available_space();
    }
    
    let usage_rate = used_disk as f64 / total_disk as f64;
    
    let message = format!(
        "磁盘：{:.1}GB / {:.1}GB (使用率：{:.1}%)",
        used_disk as f64 / 1024.0 / 1024.0 / 1024.0,
        total_disk as f64 / 1024.0 / 1024.0 / 1024.0,
        usage_rate * 100.0
    );
    
    let status = if usage_rate > 0.9 {
        "unhealthy"
    } else if usage_rate > 0.8 {
        "degraded"
    } else {
        "healthy"
    }.to_string();
    
    HealthCheckItem {
        status,
        message: Some(message),
        response_time_ms: None,
    }
}

/// 定期生成汇总数据（每日/每月）
pub async fn generate_period_summary(
    &self,
    period: &str, // "2026-03"
) -> Result<(), sea_orm::DbErr> {
    // 1. 查询该期间的所有辅助核算记录
    let records = self.find_by_period(period).await?;
    
    // 2. 按维度分组汇总
    let mut summary_map: HashMap<String, assist_accounting_summary::ActiveModel> = HashMap::new();
    
    for record in records {
        let key = format!("{}-{}-{}", 
            period, 
            record.dimension_id, 
            record.dimension_value
        );
        
        let entry = summary_map.entry(key).or_insert_with(|| {
            assist_accounting_summary::ActiveModel {
                id: Set(0),
                accounting_period: Set(period.to_string()),
                account_subject_id: Set(record.account_subject_id),
                dimension_id: Set(record.dimension_id),
                dimension_value: Set(record.dimension_value.clone()),
                initial_balance_debit: Set(Decimal::ZERO),
                initial_balance_credit: Set(Decimal::ZERO),
                current_period_debit: Set(Decimal::ZERO),
                current_period_credit: Set(Decimal::ZERO),
                ending_balance_debit: Set(Decimal::ZERO),
                ending_balance_credit: Set(Decimal::ZERO),
                created_at: Set(Utc::now()),
                updated_at: Set(Utc::now()),
            }
        });
        
        // 累加金额
        let current_debit = entry.current_period_debit.take().unwrap();
        entry.current_period_debit = Set(current_debit + record.debit_amount);
        
        let current_credit = entry.current_period_credit.take().unwrap();
        entry.current_period_credit = Set(current_credit + record.credit_amount);
    }
    
    // 3. 批量插入或更新汇总数据
    for (_, summary) in summary_map {
        // 使用 upsert 逻辑
        assist_accounting_summary::Entity::insert(summary)
            .on_conflict(
                // 唯一键冲突时更新
            )
            .exec(&self.db)
            .await?;
    }
    
    Ok(())
}

/// 校验追溯链条完整性
pub async fn validate_trace_chain(
    &self,
    trace_chain_id: &str,
) -> Result<TraceChainValidationResult, sea_orm::DbErr> {
    // 查询完整链条
    let traces = self.find_trace_chain_by_trace_chain_id(trace_chain_id).await?;
    
    let mut result = TraceChainValidationResult {
        trace_chain_id: trace_chain_id.to_string(),
        is_complete: true,
        missing_stages: Vec::new(),
        total_stages: traces.len(),
    };
    
    // 检查必需环节
    let required_stages = vec![
        "PURCHASE_RECEIPT",
        "INVENTORY_IN",
        "PRODUCTION_INPUT",
        "PRODUCTION_OUTPUT",
        "INVENTORY_OUT",
        "SALES_DELIVERY",
    ];
    
    let actual_stages: HashSet<_> = traces.iter()
        .map(|t| t.current_stage.as_str())
        .collect();
    
    for stage in required_stages {
        if !actual_stages.contains(stage) {
            result.is_complete = false;
            result.missing_stages.push(stage.to_string());
        }
    }
    
    // 检查数量一致性
    let first_stage = traces.first();
    let last_stage = traces.last();
    
    if let (Some(first), Some(last)) = (first_stage, last_stage) {
        if first.quantity_meters != last.quantity_meters {
            result.has_quantity_mismatch = true;
            result.quantity_difference = last.quantity_meters - first.quantity_meters;
        }
    }
    
    Ok(result)
}

/// 生成追溯可视化数据
pub async fn generate_trace_visualization(
    &self,
    five_dimension_id: &str,
) -> Result<TraceVisualizationNode, sea_orm::DbErr> {
    let traces = self.find_trace_chain_by_five_dimension(five_dimension_id).await?;
    
    // 构建树形结构
    let mut root: Option<TraceVisualizationNode> = None;
    let mut node_map: HashMap<i32, TraceVisualizationNode> = HashMap::new();
    
    for trace in traces {
        let node = TraceVisualizationNode {
            node_id: trace.trace_chain_id.clone(),
            node_type: trace.current_stage.clone(),
            node_name: get_stage_name(&trace.current_stage),
            bill_no: trace.current_bill_no.clone(),
            bill_date: trace.created_at,
            quantity_meters: trace.quantity_meters,
            quantity_kg: trace.quantity_kg,
            entity_name: get_entity_name(&trace),
            status: trace.trace_status.clone(),
            children: Vec::new(),
        };
        
        node_map.insert(trace.id, node);
        
        // 构建树形关系
        if let Some(prev_id) = trace.previous_trace_id {
            if let Some(parent) = node_map.get_mut(&prev_id) {
                parent.children.push(node_map.remove(&trace.id).unwrap());
            }
        } else {
            root = Some(node_map.remove(&trace.id).unwrap());
        }
    }
    
    root.ok_or_else(|| sea_orm::DbErr::RecordNotFound("追溯链为空".to_string()))
}
