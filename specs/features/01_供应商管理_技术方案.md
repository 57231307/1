## 一、技术架构

### 1.1 技术栈
- **后端框架**: Axum 0.7 + Tokio
- **ORM 框架**: SeaORM 1.0
- **数据库**: PostgreSQL 18.0
- **序列化**: Serde + Serde JSON
- **验证**: Validator
- **日期处理**: Chrono
- **精确数值**: Rust Decimal
- **认证**: JWT (jsonwebtoken)

### 1.2 架构分层

## 二、数据库设计

### 2.1 供应商表 (suppliers)

```sql
-- 供应商表
CREATE TABLE suppliers (
    id SERIAL PRIMARY KEY,
    supplier_code VARCHAR(50) NOT NULL UNIQUE,           -- 供应商编码
    supplier_name VARCHAR(200) NOT NULL UNIQUE,          -- 供应商名称
    supplier_short_name VARCHAR(100) NOT NULL,           -- 供应商简称
    supplier_type VARCHAR(50) NOT NULL,                  -- 供应商类型
    credit_code VARCHAR(50) NOT NULL UNIQUE,             -- 统一社会信用代码
    registered_address VARCHAR(500) NOT NULL,            -- 注册地址
    business_address VARCHAR(500),                       -- 经营地址
    legal_representative VARCHAR(50) NOT NULL,           -- 法人代表
    registered_capital DECIMAL(15,2) NOT NULL,           -- 注册资本（万元）
    establishment_date DATE NOT NULL,                    -- 成立日期
    business_term VARCHAR(100),                          -- 营业期限
    business_scope TEXT,                                 -- 经营范围
    taxpayer_type VARCHAR(50) NOT NULL,                  -- 纳税人类型
    bank_name VARCHAR(100) NOT NULL,                     -- 开户银行
    bank_account VARCHAR(50) NOT NULL,                   -- 银行账号
    contact_phone VARCHAR(50) NOT NULL,                  -- 联系电话
    fax VARCHAR(50),                                     -- 传真
    website VARCHAR(200),                                -- 公司网址
    email VARCHAR(100),                                  -- 联系邮箱
    main_business VARCHAR(500),                          -- 主营业务
    main_market VARCHAR(500),                            -- 主要市场
    employee_count INTEGER,                              -- 员工人数
    annual_revenue DECIMAL(15,2),                        -- 年营业额（万元）
    
    -- 等级管理
    grade VARCHAR(10) DEFAULT 'B',                       -- 供应商等级（A/B/C/D）
    grade_score DECIMAL(5,2) DEFAULT 0.00,               -- 等级评分
    last_evaluation_date DATE,                           -- 最后评估日期
    
    -- 状态管理
    status VARCHAR(20) DEFAULT 'active',                 -- 状态：active/inactive/disabled/blacklisted
    is_enabled BOOLEAN DEFAULT TRUE,                     -- 是否启用
    
    -- 辅助核算字段（面料行业特色）
    assist_batch BOOLEAN DEFAULT FALSE,                  -- 是否启用批次核算
    assist_supplier BOOLEAN DEFAULT TRUE,                -- 是否启用供应商核算
    
    -- 系统字段
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    created_by INTEGER,                                  -- 创建人 ID
    updated_by INTEGER,                                  -- 更新人 ID
    remarks TEXT                                         -- 备注
);

-- 索引
CREATE INDEX idx_suppliers_type ON suppliers(supplier_type);
CREATE INDEX idx_suppliers_grade ON suppliers(grade);
CREATE INDEX idx_suppliers_status ON suppliers(status);
CREATE INDEX idx_suppliers_credit_code ON suppliers(credit_code);
CREATE INDEX idx_suppliers_enabled ON suppliers(is_enabled);

-- 触发器：自动更新 updated_at
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_suppliers_updated_at
    BEFORE UPDATE ON suppliers
    FOR EACH ROW
    EXECUTE FUNCTION update_updated_at_column();
```

```sql
CREATE INDEX idx_supplier_qualifications_valid_until ON supplier_qualifications(valid_until);

-- 触发器：自动更新过期状态
CREATE OR REPLACE FUNCTION update_qualification_expired()
RETURNS TRIGGER AS $$
BEGIN
    IF NEW.valid_until < CURRENT_DATE THEN
        NEW.is_expired := TRUE;
    ELSE
        NEW.is_expired := FALSE;
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_supplier_qualifications_check_expired
    BEFORE INSERT OR UPDATE ON supplier_qualifications
    FOR EACH ROW
    EXECUTE FUNCTION update_qualification_expired();

-- 触发器
CREATE TRIGGER trg_supplier_qualifications_updated_at
    BEFORE UPDATE ON supplier_qualifications
    FOR EACH ROW
    EXECUTE FUNCTION update_updated_at_column();
```

### 4.1 供应商服务

```rust
pub struct SupplierService {
    db: Arc<DatabaseConnection>,
}

impl SupplierService {
    /// 创建供应商（含联系人和资质）
    pub async fn create_supplier(
        &self,
        req: CreateSupplierRequest,
        user_id: i32,
    ) -> Result<Supplier, AppError> {
        let txn = self.db.begin().await?;
        
        // 1. 生成供应商编码
        let supplier_code = self.generate_supplier_code().await?;
        
        // 2. 创建供应商
        let supplier = suppliers::ActiveModel {
            supplier_code: Set(supplier_code),
            supplier_name: Set(req.supplier_name),
            // ... 其他字段
            created_by: Set(Some(user_id)),
            ..Default::default()
        }.insert(&txn).await?;
        
        // 3. 创建联系人
        for contact_req in req.contacts {
            supplier_contacts::ActiveModel {
                supplier_id: Set(supplier.id),
                contact_name: Set(contact_req.contact_name),
                // ... 其他字段
            }.insert(&txn).await?;
        }
        
        // 4. 创建资质
        for qual_req in req.qualifications {
            supplier_qualifications::ActiveModel {
                supplier_id: Set(supplier.id),
                qualification_name: Set(qual_req.qualification_name),
                // ... 其他字段
            }.insert(&txn).await?;
        }
        
        txn.commit().await?;
        Ok(supplier)
    }
    
    /// 生成供应商编码
    async fn generate_supplier_code(&self) -> Result<String, AppError> {
        let today = Utc::now().format("%Y%m%d").to_string();
        let prefix = format!("SUP{}", today);
        
        // 查询今日已有数量
        let count = suppliers::Entity::find()
            .filter(suppliers::Column::SupplierCode.starts_with(&prefix))
            .count(&*self.db)
            .await?;
        
        Ok(format!("{}{:03}", prefix, count + 1))
    }
    
    /// 更新供应商等级
    pub async fn update_supplier_grade(
        &self,
        supplier_id: i32,
        new_score: Decimal,
    ) -> Result<(), AppError> {
        // 查询新等级
        let grade = supplier_grades::Entity::find()
            .filter(
                Condition::all()
                    .add(supplier_grades::Column::MinScore.lte(new_score))
                    .add(supplier_grades::Column::MaxScore.gt(new_score))
            )
            .one(&*self.db)
            .await?
            .ok_or(AppError::NotFound("等级未找到"))?;
        
        // 更新供应商
        let supplier = suppliers::Entity::find_by_id(supplier_id)
            .one(&*self.db)
            .await?
            .ok_or(AppError::NotFound("供应商未找到"))?;
        
        let mut supplier: suppliers::ActiveModel = supplier.into();
        supplier.grade = Set(grade.grade_code.clone());
        supplier.grade_score = Set(new_score);
        supplier.updated_at = Set(Utc::now().into());
        supplier.update(&*self.db).await?;
        
        Ok(())
    }
    
    /// 检查供应商是否可删除
    pub async fn can_delete_supplier(&self, supplier_id: i32) -> Result<bool, AppError> {
        // 检查是否有采购订单
        let order_count = purchase_orders::Entity::find()
            .filter(purchase_orders::Column::SupplierId.eq(supplier_id))
            .count(&*self.db)
            .await?;
        
        Ok(order_count == 0)
    }
}
```

```rust
        let quality_score = self.calculate_quality_score(&req)?;

        let delivery_score = self.calculate_delivery_score(&req)?;
        
        // 价格水平（20 分）
        let price_score = self.calculate_price_score(&req)?;
        
        // 服务水平（15 分）
        let service_score = self.calculate_service_score(&req)?;
        
        // 技术能力（5 分）
        let tech_score = self.calculate_tech_score(&req)?;
        
        // 总分
        let total_score = quality_score.total 
            + delivery_score.total 
            + price_score.total 
            + service_score.total 
            + tech_score.total;
        
        Ok(CalculationResult {
            quality_score,
            delivery_score,
            price_score,
            service_score,
            tech_score,
            total_score,
        })
    }
    
    /// 计算质量得分
    fn calculate_quality_score(&self, req: &SubmitScoreRequest) -> QualityScore {
        // 来料合格率（15 分）
        let pass_rate_score = if req.quality_pass_rate >= 98.0 {
            15.0
        } else if req.quality_pass_rate >= 95.0 {
            12.0
        } else if req.quality_pass_rate >= 90.0 {
            8.0
        } else {
            0.0
        };
        
        // 质量事故（10 分）
        let incident_score = match req.quality_incidents {
            0 => 10.0,
            1 => 5.0,
            _ => 0.0,
        };
        
        QualityScore {
            pass_rate_score,
            incident_score,
            improvement_score: req.quality_improvement_score,
            total: pass_rate_score + incident_score + req.quality_improvement_score,
        }
```

```rust
impl IntoResponse for SupplierError {
    fn into_response(self) -> Response {
        let (status, message) = match &self {
            SupplierError::NotFound(msg) => {
                (StatusCode::NOT_FOUND, msg.clone())
            }
            SupplierError::AlreadyExists(msg) => {
                (StatusCode::CONFLICT, msg.clone())
            }
            SupplierError::SupplierDisabled => {
                (StatusCode::BAD_REQUEST, "供应商已停用".to_string())
            }
            SupplierError::SupplierBlacklisted => {
                (StatusCode::FORBIDDEN, "供应商已列入黑名单".to_string())
            }
            SupplierError::HasTransactions => {
                (StatusCode::CONFLICT, "供应商有交易记录，无法删除".to_string())
            }
            SupplierError::QualificationExpired => {
                (StatusCode::BAD_REQUEST, "资质已过期".to_string())
            }
            SupplierError::ValidationError(msg) => {
                (StatusCode::BAD_REQUEST, msg.clone())
            }
            _ => {
                (StatusCode::INTERNAL_SERVER_ERROR, "服务器错误".to_string())
            }
        };
        
        let body = Json(serde_json::json!({
            "error": true,
            "message": message,
            "error_type": format!("{:?}", self)
        }));
        
        (status, body).into_response()
    }
}
```
