## 二、双计量单位自动换算优化

### 2.1 核心算法优化

#### 现状分析
- 当前实现：简单的米数 × 克重 = 公斤数
- 问题：未考虑幅宽、密度等因素，精度不够

#### 优化方案
```rust
/// 双计量单位换算工具（优化版）
pub struct DualUnitConverter;

impl DualUnitConverter {
    /// 米数转公斤数（精确版）
    /// 
    /// 公式：公斤数 = 米数 × 克重 (g/m²) × 幅宽 (m) ÷ 1000
    pub fn meters_to_kg(
        quantity_meters: Decimal,
        gram_weight: Decimal,  // g/m²
        width_cm: Decimal,     // cm
    ) -> Result<Decimal, String> {
        if quantity_meters < Decimal::ZERO {
            return Err("米数不能为负数".to_string());
        }
        if gram_weight <= Decimal::ZERO {
            return Err("克重必须大于 0".to_string());
        }
        if width_cm <= Decimal::ZERO {
            return Err("幅宽必须大于 0".to_string());
        }
        
        let width_m = width_cm / Decimal::from(100);
        let quantity_kg = quantity_meters * gram_weight * width_m / Decimal::from(1000);
        
        // 保留 3 位小数
        Ok(quantity_kg.round_dp(3))
    }
    
    /// 公斤数转米数（精确版）
    /// 
    /// 公式：米数 = 公斤数 × 1000 ÷ 克重 (g/m²) ÷ 幅宽 (m)
    pub fn kg_to_meters(
        quantity_kg: Decimal,
        gram_weight: Decimal,
        width_cm: Decimal,
    ) -> Result<Decimal, String> {
        if quantity_kg < Decimal::ZERO {
            return Err("公斤数不能为负数".to_string());
        }
        if gram_weight <= Decimal::ZERO {
            return Err("克重必须大于 0".to_string());
        }
        if width_cm <= Decimal::ZERO {
            return Err("幅宽必须大于 0".to_string());
        }
        
        let width_m = width_cm / Decimal::from(100);
        let quantity_meters = quantity_kg * Decimal::from(1000) / (gram_weight * width_m);
        
        // 保留 2 位小数
        Ok(quantity_meters.round_dp(2))
    }
    
    /// 自动换算（根据输入单位智能判断）
    pub fn auto_convert(
        quantity: Decimal,
        unit: &str,
        gram_weight: Decimal,
        width_cm: Decimal,
    ) -> Result<(Decimal, Decimal), String> {
        match unit {
            "米" | "M" | "meters" => {
                let kg = Self::meters_to_kg(quantity, gram_weight, width_cm)?;
                Ok((quantity, kg))
            }
            "公斤" | "KG" | "kg" => {
                let meters = Self::kg_to_meters(quantity, gram_weight, width_cm)?;
                Ok((meters, quantity))
            }
            _ => Err(format!("不支持的单位：{}", unit))
        }
    }
    
    /// 验证双计量单位一致性
    pub fn validate_dual_unit(
        quantity_meters: Decimal,
        quantity_kg: Decimal,
        gram_weight: Decimal,
        width_cm: Decimal,
        tolerance: Option<Decimal>, // 允许误差（默认 0.5%）
    ) -> Result<bool, String> {
        let calculated_kg = Self::meters_to_kg(
            quantity_meters,
            gram_weight,
            width_cm
        )?;
        
        let tolerance = tolerance.unwrap_or(Decimal::from_str("0.005").unwrap());
        let diff = (calculated_kg - quantity_kg).abs();
        let allowed_diff = calculated_kg * tolerance;
        
        Ok(diff <= allowed_diff)
    }
}
```

#### 应用场景
1. **采购入库**：自动从米数换算公斤数
2. **销售出库**：支持按米或公斤计价
3. **生产领料**：BOM 按米，实际按公斤
4. **库存盘点**：双单位同时盘点，自动校验

RETURNS TRIGGER AS $$
BEGIN
    IF NEW.quantity_meters IS NOT NULL 
       AND NEW.gram_weight IS NOT NULL 
       AND NEW.width_cm IS NOT NULL THEN
        NEW.quantity_kg := ROUND(
            NEW.quantity_meters * NEW.gram_weight * (NEW.width_cm / 100.0) / 1000.0,
            3
        );
        NEW.calculated_quantity_kg := NEW.quantity_kg;
        NEW.unit_conversion_rate := ROUND(
            NEW.quantity_kg / NULLIF(NEW.quantity_meters, 0),
            6
        );
    END IF;
    RETURN NEW;
END;

#[derive(Debug, Deserialize)]
pub struct ConvertUnitRequest {

/// 双计量单位换算响应
#[derive(Debug, Serialize)]
pub struct ConvertUnitResponse {
    pub original_quantity: f64,
    pub original_unit: String,
    pub converted_quantity: f64,
    pub converted_unit: String,
    pub conversion_rate: f64,
    pub formula: String,

/// 双计量单位换算接口
pub async fn convert_dual_unit(
    State(db): State<Arc<DatabaseConnection>>,
    Json(req): Json<ConvertUnitRequest>,
) -> impl IntoResponse {
    use rust_decimal::prelude::ToPrimitive;
    
    // 查询产品的克重和幅宽
    let product = Product::find_by_id(req.product_id)
        .one(&db)
        .await
        .map_err(|e| (StatusCode::INTERNAL_SERVER_ERROR, e.to_string()))?
        .ok_or((StatusCode::NOT_FOUND, "产品不存在"))?;
    
    let gram_weight = product.gram_weight
        .ok_or((StatusCode::BAD_REQUEST, "产品未设置克重"))?;
    let width = product.width
        .ok_or((StatusCode::BAD_REQUEST, "产品未设置幅宽"))?;
    
    let quantity = Decimal::from_f64_retain(req.quantity)
        .unwrap_or(Decimal::ZERO);
    
    let (meters, kg, formula) = match req.from_unit.as_str() {
        "米" | "M" | "meters" => {
            let kg = DualUnitConverter::meters_to_kg(
                quantity,
                gram_weight,
                width
            ).map_err(|e| (StatusCode::BAD_REQUEST, e))?;
            let formula = format!(
                "{}米 × {}g/m² × {}m ÷ 1000 = {}公斤",
                req.quantity,
                gram_weight,
                width / Decimal::from(100),
                kg
            );
            (quantity, kg, formula)
        }
        "公斤" | "KG" | "kg" => {
            let meters = DualUnitConverter::kg_to_meters(
                quantity,
                gram_weight,
                width
            ).map_err(|e| (StatusCode::BAD_REQUEST, e))?;
            let formula = format!(
                "{}公斤 × 1000 ÷ {}g/m² ÷ {}m = {}米",
                req.quantity,
                gram_weight,
                width / Decimal::from(100),
                meters
            );
            (meters, quantity, formula)
        }
        _ => return Err((StatusCode::BAD_REQUEST, "不支持的单位"))
    };
    
    Ok(Json(ConvertUnitResponse {
        original_quantity: req.quantity,
        original_unit: req.from_unit,
        converted_quantity: if req.from_unit == "米" { 
            kg.to_f64().unwrap() 
        } else { 
            meters.to_f64().unwrap() 
        },
        converted_unit: if req.from_unit == "米" { 
            "公斤".to_string() 
        } else { 
            "米".to_string() 
        },
        conversion_rate: (kg / meters).to_f64().unwrap(),
        formula,
    }))
}

/// 面料行业五维管理核心结构
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct FabricFiveDimension {
    /// 第一维：成品 ID
    pub product_id: i32,
    
    /// 第二维：批次号（坯布批次/成品批次）
    pub batch_no: String,
    
    /// 第三维：色号（潘通色号/定制色号）
    pub color_no: String,
    
    /// 第四维：缸号（染缸编号）
    pub dye_lot_no: String,
    
    /// 第五维：等级（一等品/二等品/等外品）
    pub grade: String,
}

impl FabricFiveDimension {
    /// 生成全局唯一标识
    pub fn unique_id(&self) -> String {
        format!(
            "P{}|B{}|C{}|D{}|G{}",
            self.product_id,
            self.batch_no,
            self.color_no,
            self.dye_lot_no,
            self.grade
        )
    }
    
    /// 从字符串解析
    pub fn from_unique_id(id: &str) -> Result<Self, String> {
        let parts: Vec<&str> = id.split('|').collect();
        if parts.len() != 5 {
            return Err("五维 ID 格式错误".to_string());
        }
        
        let product_id = parts[0][1..].parse::<i32>()
            .map_err(|_| "产品 ID 解析失败")?;
        
        Ok(Self {
            product_id,
            batch_no: parts[1][1..].to_string(),
            color_no: parts[2][1..].to_string(),
            dye_lot_no: parts[3][1..].to_string(),
            grade: parts[4][1..].to_string(),
        })
    }
    
    /// 验证五维数据的完整性
    pub fn validate(&self) -> Result<(), String> {
        if self.product_id <= 0 {
            return Err("产品 ID 必须大于 0".to_string());
        }
        if self.batch_no.is_empty() {
            return Err("批次号不能为空".to_string());
        }
        if self.color_no.is_empty() {
            return Err("色号不能为空".to_string());
        }
        if self.dye_lot_no.is_empty() {
            return Err("缸号不能为空".to_string());
        }
        if self.grade.is_empty() {
            return Err("等级不能为空".to_string());
        }
        Ok(())
    }
}

/// 正向追溯：供应商 → 采购 → 库存 → 生产 → 销售 → 客户
pub async fn forward_trace(
    db: &DatabaseConnection,
    five_dimension: &FabricFiveDimension,
) -> Result<BusinessTraceChain, DbErr> {
    let mut trace_nodes = Vec::new();
    
    // 1. 查询采购入库（供应商来源）
    let purchase_receipts = PurchaseReceiptItem::find()
        .filter(purchase_receipt_item::Column::BatchNo.eq(&five_dimension.batch_no))
        .filter(purchase_receipt_item::Column::ColorNo.eq(&five_dimension.color_no))
        .filter(purchase_receipt_item::Column::DyeLotNo.eq(&five_dimension.dye_lot_no))
        .all(db)
        .await?;
    
    for receipt in purchase_receipts {
        trace_nodes.push(TraceNode {
            node_type: NodeType::PurchaseReceipt,
            document_type: "采购入库单".to_string(),
            document_no: receipt.receipt_no,
            document_date: receipt.created_at,
            quantity_meters: receipt.quantity_received,
            quantity_kg: receipt.quantity_kg,
            previous_node_id: None,
            remarks: None,
        });
    }
    
    // 2. 查询库存
    let stocks = InventoryStock::find()
        .filter(inventory_stock::Column::BatchNo.eq(&five_dimension.batch_no))
        .filter(inventory_stock::Column::ColorNo.eq(&five_dimension.color_no))
        .filter(inventory_stock::Column::DyeLotNo.eq(&five_dimension.dye_lot_no))
        .all(db)
        .await?;
    
    for stock in stocks {
        trace_nodes.push(TraceNode {
            node_type: NodeType::Inventory,
            document_type: "库存".to_string(),
            document_no: format!("库存 ID:{}", stock.id),
            document_date: stock.created_at,
            quantity_meters: stock.quantity_meters,
            quantity_kg: stock.quantity_kg,
            previous_node_id: None,
            remarks: None,
        });
    }
    
    // 3. 查询生产领用
    let production_usage = ProductionMaterialUsage::find()
        .filter(production_material_usage::Column::BatchNo.eq(&five_dimension.batch_no))
        .filter(production_material_usage::Column::ColorNo.eq(&five_dimension.color_no))
        .all(db)
        .await?;
    
    for usage in production_usage {
        trace_nodes.push(TraceNode {
            node_type: NodeType::Production,
            document_type: "生产领料单".to_string(),
            document_no: usage.usage_no,
            document_date: usage.usage_date,
            quantity_meters: usage.quantity_used,
            quantity_kg: usage.quantity_kg,
            previous_node_id: None,
            remarks: None,
        });
    }
    
    // 4. 查询销售出库
    let sales_deliveries = SalesDeliveryItem::find()
        .filter(sales_delivery_item::Column::BatchNo.eq(&five_dimension.batch_no))
        .filter(sales_delivery_item::Column::ColorNo.eq(&five_dimension.color_no))
        .all(db)
        .await?;
    
    for delivery in sales_deliveries {
        trace_nodes.push(TraceNode {
            node_type: NodeType::SalesDelivery,
            document_type: "销售出库单".to_string(),
            document_no: delivery.delivery_no,
            document_date: delivery.created_at,
            quantity_meters: delivery.quantity,
            quantity_kg: delivery.quantity_kg,
            previous_node_id: None,
            remarks: None,
        });
    }
    
    Ok(BusinessTraceChain {
        trace_type: TraceType::Forward,
        five_dimension: five_dimension.clone(),
        trace_nodes,
    })
}
