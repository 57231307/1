```rust
/// 计算评估总分（100 分制）
pub fn calculate_total_score(
    &self,
    scores: &EvaluationScores,
) -> Decimal {
    // 质量水平（35 分）
    let quality_score = self.calculate_quality_score(scores);
    
    // 交货能力（25 分）
    let delivery_score = self.calculate_delivery_score(scores);
    
    // 价格水平（20 分）
    let price_score = self.calculate_price_score(scores);
    
    // 服务水平（15 分）
    let service_score = self.calculate_service_score(scores);
    
    // 技术能力（5 分）
    let tech_score = self.calculate_tech_score(scores);
    
    quality_score + delivery_score + price_score + service_score + tech_score
}
```

#### 智能等级判定
```rust
/// 根据总分确定等级
pub async fn determine_grade(&self, total_score: Decimal) -> Result<String, AppError> {
    let grades = supplier_grade::Entity::find()
        .filter(supplier_grade::Column::IsEnabled.eq(true))
        .all(&*self.db)
        .await?;

    for grade in grades {
        if total_score >= grade.min_score && total_score <= grade.max_score {
            return Ok(grade.grade_code);
        }
    }

    Ok("D".to_string()) // 默认 D 级
}
```

#### 自动降级预警
```rust
/// 检查是否需要降级
pub async fn check_grade_downgrade(
    &self,
    supplier_id: i32,
    new_score: Decimal,
) -> Result<bool, AppError> {
    let supplier = supplier::Entity::find_by_id(supplier_id)
        .one(&*self.db)
        .await?;

    let current_grade = &supplier.grade;
    let grade = supplier_grade::Entity::find()
        .filter(supplier_grade::Column::GradeCode.eq(current_grade))
        .one(&*self.db)
        .await?;

    Ok(new_score < grade.min_score) // 低于当前等级最低分，需要降级
}
```

#### 黑名单自动检测
```rust
/// 检查是否需要列入黑名单
pub async fn check_blacklist(&self, supplier_id: i32) -> Result<bool, AppError> {
    // 查询最近 3 次评估
    let evaluations = supplier_evaluation::Entity::find()
        .filter(supplier_evaluation::Column::SupplierId.eq(supplier_id))
        .order_by_desc(supplier_evaluation::Column::EvaluationYear)
        .order_by_desc(supplier_evaluation::Column::EvaluationQuarter)
        .limit(3)
        .all(&*self.db)
        .await?;

    // 连续 3 次 D 级，自动列入黑名单
    if evaluations.len() >= 3 {
        let all_d_grade = evaluations.iter().all(|e| {
            e.grade_after.as_ref().map_or(false, |g| g == "D")
        });

        if all_d_grade {
            return Ok(true);
        }
    }

    Ok(false)
}
```

```rust
/// 评估分数结构（11 个维度）
#[derive(Debug, Clone)]
pub struct EvaluationScores {
    // 质量水平（35 分）
    pub quality_pass_rate: Decimal,        // 来料合格率（%）
    pub quality_incidents: i32,            // 质量事故次数
    pub quality_improvement_score: Decimal, // 质量改进得分

    // 交货能力（25 分）
    pub delivery_on_time_rate: Decimal,    // 交货及时率（%）
    pub order_completion_rate: Decimal,    // 订单完成率（%）

    // 价格水平（20 分）
    pub price_competitiveness_score: Decimal, // 价格竞争力得分
    pub price_stability_score: Decimal,    // 价格稳定性得分

    // 服务水平（15 分）
    pub response_speed_score: Decimal,     // 响应速度得分
    pub after_sales_score: Decimal,        // 售后服务得分
    pub cooperation_score: Decimal,        // 配合度得分

    // 技术能力（5 分）
    pub rd_capability_score: Decimal,      // 研发能力得分
    pub technical_support_score: Decimal,  // 技术支持得分
}
```
