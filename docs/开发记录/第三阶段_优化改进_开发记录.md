```rust
/// 生成会计期间汇总（按月）- 完整实现
pub async fn generate_monthly_summary(
    &self,
    year: i32,
    month: u32,
) -> Result<(), sea_orm::DbErr> {
    use sea_orm::{QuerySelect, QueryOrder};
    
    let accounting_period = format!("{:04}-{:02}", year, month);

    // 1. 查询该月的所有辅助核算记录
    let start_date = format!("{:04}-{:02}-01 00:00:00", year, month);
    let end_date = format!("{:04}-{:02}-31 23:59:59", year, month);
    
    let records = assist_accounting_record::Entity::find()
        .filter(assist_accounting_record::Column::CreatedAt.gte(start_date))
        .filter(assist_accounting_record::Column::CreatedAt.lte(end_date))
        .all(&self.db)
        .await?;

    // 2. 按维度、科目分组汇总
    use std::collections::HashMap;
    let mut summary_map: HashMap<String, assist_accounting_summary::ActiveModel> = HashMap::new();

    for record in records {
        // 按批次维度汇总
        let batch_key = format!("{}-BATCH-{}-{}", accounting_period, record.batch_no, record.account_subject_id);
        self.update_summary_map(&mut summary_map, &batch_key, &accounting_period, "BATCH", &record.batch_no, record.account_subject_id, record.debit_amount, record.credit_amount, record.quantity_meters, record.quantity_kg);

        // 按色号维度汇总
        let color_key = format!("{}-COLOR-{}-{}", accounting_period, record.color_no, record.account_subject_id);
        self.update_summary_map(&mut summary_map, &color_key, &accounting_period, "COLOR", &record.color_no, record.account_subject_id, record.debit_amount, record.credit_amount, record.quantity_meters, record.quantity_kg);

        // 按仓库维度汇总
        let warehouse_key = format!("{}-WAREHOUSE-{}-{}", accounting_period, record.warehouse_id, record.account_subject_id);
        self.update_summary_map(&mut summary_map, &warehouse_key, &accounting_period, "WAREHOUSE", &record.warehouse_id.to_string(), record.account_subject_id, record.debit_amount, record.credit_amount, record.quantity_meters, record.quantity_kg);
    }

    // 3. 批量插入或更新汇总数据
    for (_, mut summary) in summary_map {
        summary.insert(&self.db).await?;
    }

    Ok(())
}

/// 更新汇总 Map（辅助函数）
fn update_summary_map(
    &self,
    summary_map: &mut HashMap<String, assist_accounting_summary::ActiveModel>,
    key: &str,
    accounting_period: &str,
    dimension_code: &str,
    dimension_value: &str,
    account_subject_id: i32,
    debit_amount: Decimal,
    credit_amount: Decimal,
    quantity_meters: Decimal,
    quantity_kg: Decimal,
) {
    use sea_orm::Set;
    
    let entry = summary_map.entry(key.to_string()).or_insert_with(|| {
        assist_accounting_summary::ActiveModel {
            id: Set(0),
            accounting_period: Set(accounting_period.to_string()),
            dimension_code: Set(dimension_code.to_string()),
            dimension_value: Set(dimension_value.to_string()),
            account_subject_id: Set(account_subject_id),
            initial_balance_debit: Set(Decimal::ZERO),
            initial_balance_credit: Set(Decimal::ZERO),
            current_period_debit: Set(Decimal::ZERO),
            current_period_credit: Set(Decimal::ZERO),
            ending_balance_debit: Set(Decimal::ZERO),
            ending_balance_credit: Set(Decimal::ZERO),
            total_quantity_meters: Set(Decimal::ZERO),
            total_quantity_kg: Set(Decimal::ZERO),
            record_count: Set(0),
            created_at: Set(chrono::Utc::now()),
            updated_at: Set(chrono::Utc::now()),
        }
    });

    // 累加金额和数量
    let current_debit = entry.current_period_debit.take().unwrap();
    entry.current_period_debit = Set(current_debit + debit_amount);

    let current_credit = entry.current_period_credit.take().unwrap();

    let current_qty_kg = entry.total_quantity_kg.take().unwrap();
    entry.total_quantity_kg = Set(current_qty_kg + quantity_kg);

    let current_count = entry.record_count.take().unwrap();
    entry.record_count = Set(current_count + 1);
}
```

```rust
/// 校验追溯链条完整性
pub async fn validate_trace_chain(
    &self,
    trace_chain_id: &str,
) -> Result<TraceChainValidationResult, sea_orm::DbErr> {
    // 查询完整链条
    let traces = self.find_trace_chain_by_id(trace_chain_id).await?;
    
    let mut result = TraceChainValidationResult {
        trace_chain_id: trace_chain_id.to_string(),
        is_complete: true,
        missing_stages: Vec::new(),
        total_stages: traces.len(),
        has_quantity_mismatch: false,
        quantity_difference: Decimal::ZERO,
        validation_time: Utc::now(),
    };
    
    if traces.is_empty() {
        result.is_complete = false;
        return Ok(result);
    }

    // 检查必需环节
    let required_stages = vec![
        "PURCHASE_RECEIPT",      // 采购收货
        "INVENTORY_IN",          // 入库
        "PRODUCTION_INPUT",      // 生产投入
        "PRODUCTION_OUTPUT",     // 生产产出
        "INVENTORY_OUT",         // 出库
        "SALES_DELIVERY",        // 销售发货
    ];
    
    let actual_stages: HashSet<_> = traces.iter()
        .map(|t| t.current_stage.as_str())
        .collect();
    
    for stage in required_stages {
        if !actual_stages.contains(stage) {
            result.is_complete = false;
            result.missing_stages.push(stage.to_string());
        }
    }
    
    // 检查数量一致性（首尾环节）
    let first_trace = traces.first();
    let last_trace = traces.last();
    
    if let (Some(first), Some(last)) = (first_trace, last_trace) {
        if first.quantity_meters != last.quantity_meters {
            result.has_quantity_mismatch = true;
            result.quantity_difference = last.quantity_meters - first.quantity_meters;
        }
    }
    
    // 检查环节连续性（是否有断链）
    let mut prev_id: Option<i32> = None;
    for trace in traces.iter() {
        if let Some(prev) = prev_id {
            if trace.previous_trace_id != Some(prev) {
                result.is_complete = false;
                result.missing_stages.push(format!("环节 {} 断裂", trace.id));
            }
        }
        prev_id = Some(trace.id);
    }
    
    Ok(result)
}
```

**说明**:
- 检查 6 个必需环节是否完整
- 检查首尾环节数量一致性
- 检查环节连续性（防止断链）
- 返回详细的校验结果

```rust
        .all(&self.db)
        .await?;
    
    // 按追溯链 ID 分组
    let mut chain_map: HashMap<String, Vec<business_trace::Model>> = HashMap::new();
    for trace in traces {
        chain_map.entry(trace.trace_chain_id.clone())
            .or_insert_with(Vec::new)
            .push(trace);
    }
    
    // 为每个追溯链生成快照
    let mut snapshot_count = 0;
    for (chain_id, chain_traces) in chain_map {
        // 检查是否已有快照
        let existing_snapshot = business_trace_snapshot::Entity::find()
            .filter(business_trace_snapshot::Column::TraceChainId.eq(&chain_id))
            .one(&self.db)
            .await?;
        
        if existing_snapshot.is_none() {
            // 没有快照则创建
            self.create_snapshot(&chain_id).await?;
            snapshot_count += 1;
        }
    }
    
    Ok(snapshot_count)
}
```

**说明**:
- 批量处理指定天数前的追溯链
- 避免重复生成快照
- 返回生成的快照数量

```rust
/// 生成追溯可视化数据（树形结构）
pub async fn generate_trace_visualization(
    &self,
    five_dimension_id: &str,
) -> Result<TraceVisualizationNode, sea_orm::DbErr> {
    let traces = self.find_trace_chain_by_five_dimension(five_dimension_id).await?;
    
    if traces.is_empty() {
        return Err(sea_orm::DbErr::RecordNotFound("追溯链为空".to_string()));
    }
    
    // 构建树形结构
    let mut node_map: HashMap<i32, TraceVisualizationNode> = HashMap::new();
    let mut root_id: Option<i32> = None;
    
    for trace in traces {
        let node = TraceVisualizationNode {
            node_id: trace.id.to_string(),
            node_type: trace.current_stage.clone(),
            node_name: Self::get_stage_name(&trace.current_stage),
            bill_no: trace.current_bill_no.clone(),
            bill_type: trace.current_bill_type.clone(),
            bill_date: trace.created_at,
            quantity_meters: trace.quantity_meters,
            quantity_kg: trace.quantity_kg,
            entity_name: Self::get_entity_name(&trace),
            status: trace.trace_status.clone(),
            children: Vec::new(),
        };
        
        node_map.insert(trace.id, node);
        
        // 构建树形关系
        if let Some(prev_id) = trace.previous_trace_id {
            if let Some(parent) = node_map.get_mut(&prev_id) {
                let child = node_map.remove(&trace.id).unwrap();
                parent.children.push(child);
            }
        } else {
            root_id = Some(trace.id);
        }
    }
    
    // 返回根节点
    root_id.and_then(|id| node_map.remove(&id))
        .ok_or_else(|| sea_orm::DbErr::RecordNotFound("追溯链根节点不存在".to_string()))
}
```

**说明**:
- 生成树形结构的追溯节点
- 支持父子关系展示
- 包含中文环节名称
- 适合前端可视化展示
