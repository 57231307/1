**导出产品列表**:
```rust
pub fn export_products_to_excel(
    products: Vec<product::Model>,
    output_path: &str,
) -> Result<(), XlsxError> {
    let mut workbook = Workbook::new();
    let mut worksheet = Worksheet::new();

    // 设置表头
    worksheet.write_string(0, 0, "产品 ID")?;
    worksheet.write_string(0, 1, "产品名称")?;
    worksheet.write_string(0, 2, "产品编码")?;
    worksheet.write_string(0, 3, "类别")?;
    worksheet.write_string(0, 4, "规格")?;
    worksheet.write_string(0, 5, "单位")?;
    worksheet.write_string(0, 6, "标准价格")?;
    worksheet.write_string(0, 7, "成本价格")?;
    worksheet.write_string(0, 8, "状态")?;

    // 写入数据
    for (idx, product) in products.iter().enumerate() {
        let row = (idx + 1) as u32;
        worksheet.write_number(row, 0, product.id as f64)?;
        worksheet.write_string(row, 1, &product.name)?;
        worksheet.write_string(row, 2, &product.code)?;
        // ... 其他字段
    }

    workbook.add_worksheet(&worksheet);
    workbook.save(output_path)?;

    Ok(())
}
```

## 四、Excel 导入功能

### 4.1 导入服务实现

**导入产品**:
```rust
pub fn import_products_from_excel(
    file_path: &str,
) -> Result<Vec<BatchCreateProductRequest>, Box<dyn std::error::Error>> {
    let mut products = Vec::new();

    // 读取 Excel 文件
    let workbook = open_workbook(file_path)?;

    if let Some(range) = workbook.worksheet_range("Sheet1")? {
        for row in range.rows().skip(1) { // 跳过表头
            if row.len() >= 2 {
                let product = BatchCreateProductRequest {
                    name: row[0].to_string(),
                    code: row[1].to_string(),
                    category_id: row.get(2).and_then(|c| c.as_i64()).map(|i| i as i32),
                    specification: row.get(3).map(|s| s.to_string()),
                    unit: row.get(4).map(|s| s.to_string()),
                    standard_price: row.get(5).and_then(|p| p.as_f64()).map(|p| p.to_string()),
                    cost_price: row.get(6).and_then(|p| p.as_f64()).map(|p| p.to_string()),
                    description: row.get(7).map(|s| s.to_string()),
                };
                products.push(product);
            }
        }
    }

    Ok(products)
}
```

    }

    /// 发送库存预警通知
    pub async fn send_low_stock_alert(
        &self,
        product_name: &str,
        current_quantity: Decimal,
        min_stock: Decimal,
    ) -> Result<(), Box<dyn std::error::Error>> {
        let title = format!("库存预警：{}", product_name);
        let content = format!(

        self.send_internal_message(
            1, // 仓库管理员 ID
            &title,
            &content,
            "low_stock_alert",
        ).await?;

        Ok(())
    }
}
